634qbool GL_CheckExtension(const char *name, const char *disableparm, int silent)
635{
636 int failed = false;
637 const glfunction_t *func;
638 char extstr[MAX_INPUTLINE];
639
640 Con_DPrintf("checking for %s... ", name);
641
642 if (disableparm && (Sys_CheckParm(disableparm) || Sys_CheckParm("-safe")))
643 {
644 Con_DPrint("disabled by commandline\n");
645 return false;
646 }
647
648 if (!GL_ExtensionSupported(name))
649 {
650 Con_DPrint("not detected\n");
651 return false;
652 }
653
654#ifndef USE_GLES2
655 for (func = openglfuncs; func && func->name != NULL; func++)
656 {
657 if (!*func->funcvariable && !strcmp(name, func->extension))
658 {
659 if (!silent)
660 Con_DPrintf("%s is missing function \"%s\" - broken driver!\n", name, func->name);
661 failed = true;
662 }
663 }
664#endif //USE_GLES2
665 // delay the return so it prints all missing functions
666 if (failed)
667 return false;
668 // VorteX: add to found extension list
669 dpsnprintf(extstr, sizeof(extstr), "%s %s ", gl_info_extensions.string, name);
670 Cvar_SetQuick(&gl_info_extensions, extstr);
671
672 Con_DPrint("enabled\n");
673 return true;
674}

855float VID_JoyState_GetAxis(const vid_joystate_t *joystate, int axis, float fsensitivity, float deadzone)
856{
857 float value;
858 value = (axis >= 0 && axis < MAXJOYAXIS) ? joystate->axis[axis] : 0.0f;
859 value = value > deadzone ? (value - deadzone) : (value < -deadzone ? (value + deadzone) : 0.0f);
860 value *= deadzone > 0 ? (1.0f / (1.0f - deadzone)) : 1.0f;
861 value = bound(-1, value, 1);
862 return value * fsensitivity;
863}

1118void VID_BuildGammaTables(unsigned short *ramps, int rampsize)

1124 BuildGammaTable16(1.0f, invpow(0.5, 1 - cachegrey[2]), cachewhite[2], cacheblack[2], cachecontrastboost, ramps + rampsize*2, rampsize);
1125 }
1126 else
1127 {

1129 BuildGammaTable16(1.0f, cachegamma, cachecontrast, cachebrightness, cachecontrastboost, ramps + rampsize, rampsize);
1130 BuildGammaTable16(1.0f, cachegamma, cachecontrast, cachebrightness, cachecontrastboost, ramps + rampsize*2, rampsize);
1131 }
1132
1133 if(vid.sRGB2D || vid.sRGB3D)
1134 {
1135 int i;
1136 for(i = 0; i < 3*rampsize; ++i)
1137 ramps[i] = (int)floor(bound(0.0f, Image_sRGBFloatFromLinearFloat(ramps[i] / 65535.0f), 1.0f) * 65535.0f + 0.5f);
1138 }

1140 // LadyHavoc: this code came from Ben Winslow and Zinx Verituse, I have
1141 // immensely butchered it to work with variable framerates and fit in with
1142 // the rest of darkplaces.
1143 //
1144 // R.I.P. zinx http://obits.al.com/obituaries/birmingham/obituary.aspx?n=christopher-robert-lais&pid=186080667
1145 if (v_psycho.integer)
1146 {
1147 int x, y;
1148 float t;
1149 static float n[3], nd[3], nt[3];
1150 static int init = true;
1151 unsigned short *ramp;
1152 gamma_forcenextframe = true;
1153 if (init)
1154 {
1155 init = false;
1156 for (x = 0;x < 3;x++)
1157 {
1158 n[x] = lhrandom(0, 1);
1159 nd[x] = (rand()&1)?-0.25:0.25;

1161 }
1162 }

1166 nt[x] -= cl.realframetime;
1167 if (nt[x] < 0)
1168 {
1169 nd[x] = -nd[x];
1170 nt[x] += lhrandom(1, 8.2);
1171 }
1172 n[x] += nd[x] * cl.realframetime;
1173 n[x] -= floor(n[x]);
1174 }

1176 for (x = 0, ramp = ramps;x < 3;x++)
1177 for (y = 0, t = n[x] - 0.75f;y < rampsize;y++, t += 0.75f * (2.0f / rampsize))
