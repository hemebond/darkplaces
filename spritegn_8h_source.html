34// This program generates .spr sprite package files.

36//
37// dsprite_t file header structure
38// <repeat dsprite_t.numframes times>

40// dspriteframe_t frame header structure
41// sprite bitmap
42// <else (single sprite frame)>

44// sprite bitmap
45// <endrepeat>
