60static void Mod_SpriteSetupTexture(texture_t *texture, skinframe_t *skinframe, qbool fullbright, qbool additive)
61{
62 if (!skinframe)
63 skinframe = R_SkinFrame_LoadMissing();
64 texture->offsetmapping = OFFSETMAPPING_OFF;
65 texture->offsetscale = 1;
66 texture->offsetbias = 0;
67 texture->specularscalemod = 1;
68 texture->specularpowermod = 1;
69 texture->basematerialflags = MATERIALFLAG_WALL;
70 texture->basealpha = 1.0f;
71 if (fullbright)
72 texture->basematerialflags |= MATERIALFLAG_FULLBRIGHT;
73 if (additive)
74 texture->basematerialflags |= MATERIALFLAG_ADD | MATERIALFLAG_BLENDED | MATERIALFLAG_NOSHADOW;
75 else if (skinframe->hasalpha)
76 texture->basematerialflags |= MATERIALFLAG_ALPHA | MATERIALFLAG_BLENDED | MATERIALFLAG_NOSHADOW;
77 texture->currentmaterialflags = texture->basematerialflags;
78 texture->materialshaderpass = texture->shaderpasses[0] = Mod_CreateShaderPass(loadmodel->mempool, skinframe);
79 texture->currentskinframe = skinframe;
80 texture->surfaceflags = 0;
81 texture->supercontents = SUPERCONTENTS_SOLID;
82 if (!(texture->basematerialflags & MATERIALFLAG_BLENDED))
83 texture->supercontents |= SUPERCONTENTS_OPAQUE;
84 texture->transparentsort = TRANSPARENTSORT_DISTANCE;
85 // WHEN ADDING DEFAULTS HERE, REMEMBER TO PUT DEFAULTS IN ALL LOADERS
86 // JUST GREP FOR "specularscalemod = 1".
87}

91static void Mod_Sprite_SharedSetup(const unsigned char *datapointer, int version, const unsigned int *palette, qbool additive)
92{
93 int i, j, groupframes, realframes, x, y, origin[2], width, height;
94 qbool fullbright;
95 dspriteframetype_t *pinframetype;
96 dspriteframe_t *pinframe;
97 dspritegroup_t *pingroup;
98 dspriteinterval_t *pinintervals;
99 skinframe_t *skinframe;
100 float modelradius, interval;
101 char name[MAX_QPATH], fogname[MAX_QPATH];
102 const void *startframes;
103 int texflags = (r_mipsprites.integer ? TEXF_MIPMAP : 0) | ((gl_texturecompression.integer && gl_texturecompression_sprites.integer) ? TEXF_COMPRESS : 0) | TEXF_ISSPRITE | TEXF_PICMIP | TEXF_ALPHA | TEXF_CLAMP;
104 modelradius = 0;
105
106 if (loadmodel->numframes < 1)
107 Host_Error ("Mod_Sprite_SharedSetup: Invalid # of frames: %d", loadmodel->numframes);
108
109 // LadyHavoc: hack to allow sprites to be non-fullbright
110 fullbright = true;
111 for (i = 0;i < MAX_QPATH && loadmodel->name[i];i++)
112 if (loadmodel->name[i] == '!')
113 fullbright = false;
114
115//
116// load the frames
117//
118 startframes = datapointer;
119 realframes = 0;
120 for (i = 0;i < loadmodel->numframes;i++)
121 {
122 pinframetype = (dspriteframetype_t *)datapointer;
123 datapointer += sizeof(dspriteframetype_t);
124
125 if (LittleLong (pinframetype->type) == SPR_SINGLE)
126 groupframes = 1;
127 else
128 {
129 pingroup = (dspritegroup_t *)datapointer;
130 datapointer += sizeof(dspritegroup_t);
131
132 groupframes = LittleLong(pingroup->numframes);
133
134 datapointer += sizeof(dspriteinterval_t) * groupframes;
135 }
136
137 for (j = 0;j < groupframes;j++)
138 {
139 pinframe = (dspriteframe_t *)datapointer;
140 if (version == SPRITE32_VERSION)
141 datapointer += sizeof(dspriteframe_t) + LittleLong(pinframe->width) * LittleLong(pinframe->height) * 4;
142 else //if (version == SPRITE_VERSION || version == SPRITEHL_VERSION)
143 datapointer += sizeof(dspriteframe_t) + LittleLong(pinframe->width) * LittleLong(pinframe->height);
144 }
145 realframes += groupframes;
146 }
147
148 loadmodel->animscenes = (animscene_t *)Mem_Alloc(loadmodel->mempool, sizeof(animscene_t) * loadmodel->numframes);
149 loadmodel->sprite.sprdata_frames = (mspriteframe_t *)Mem_Alloc(loadmodel->mempool, sizeof(mspriteframe_t) * realframes);
150 loadmodel->num_textures = realframes;
151 loadmodel->num_texturesperskin = 1;
152 loadmodel->data_textures = (texture_t *)Mem_Alloc(loadmodel->mempool, sizeof(texture_t) * loadmodel->num_textures);
153
154 datapointer = (unsigned char *)startframes;
155 realframes = 0;
156 for (i = 0;i < loadmodel->numframes;i++)
157 {
158 pinframetype = (dspriteframetype_t *)datapointer;
159 datapointer += sizeof(dspriteframetype_t);
160
161 if (LittleLong (pinframetype->type) == SPR_SINGLE)
162 {
163 groupframes = 1;
164 interval = 0.1f;
165 }
166 else
167 {
168 pingroup = (dspritegroup_t *)datapointer;
169 datapointer += sizeof(dspritegroup_t);
170
171 groupframes = LittleLong(pingroup->numframes);
172
173 pinintervals = (dspriteinterval_t *)datapointer;
174 datapointer += sizeof(dspriteinterval_t) * groupframes;
175
176 interval = LittleFloat(pinintervals[0].interval);
177 if (interval < 0.01f)
178 Host_Error("Mod_Sprite_SharedSetup: invalid interval");
179 }
180
181 dpsnprintf(loadmodel->animscenes[i].name, sizeof(loadmodel->animscenes[i].name), "frame %i", i);
182 loadmodel->animscenes[i].firstframe = realframes;
183 loadmodel->animscenes[i].framecount = groupframes;
184 loadmodel->animscenes[i].framerate = 1.0f / interval;
185 loadmodel->animscenes[i].loop = true;
186
187 for (j = 0;j < groupframes;j++)
188 {
189 pinframe = (dspriteframe_t *)datapointer;
190 datapointer += sizeof(dspriteframe_t);
191
192 origin[0] = LittleLong (pinframe->origin[0]);
193 origin[1] = LittleLong (pinframe->origin[1]);
194 width = LittleLong (pinframe->width);
195 height = LittleLong (pinframe->height);
196
197 loadmodel->sprite.sprdata_frames[realframes].left = origin[0];
198 loadmodel->sprite.sprdata_frames[realframes].right = origin[0] + width;
199 loadmodel->sprite.sprdata_frames[realframes].up = origin[1];
200 loadmodel->sprite.sprdata_frames[realframes].down = origin[1] - height;
201
202 x = (int)max(loadmodel->sprite.sprdata_frames[realframes].left * loadmodel->sprite.sprdata_frames[realframes].left, loadmodel->sprite.sprdata_frames[realframes].right * loadmodel->sprite.sprdata_frames[realframes].right);
203 y = (int)max(loadmodel->sprite.sprdata_frames[realframes].up * loadmodel->sprite.sprdata_frames[realframes].up, loadmodel->sprite.sprdata_frames[realframes].down * loadmodel->sprite.sprdata_frames[realframes].down);
204 if (modelradius < x + y)
205 modelradius = x + y;
206
207 if (cls.state != ca_dedicated)
208 {
209 skinframe = NULL;
210 // note: Nehahra's null.spr has width == 0 and height == 0
211 if (width > 0 && height > 0)
212 {
213 if (groupframes > 1)
214 {
215 dpsnprintf (name, sizeof(name), "%s_%i_%i", loadmodel->name, i, j);
216 dpsnprintf (fogname, sizeof(fogname), "%s_%i_%ifog", loadmodel->name, i, j);
217 }
218 else
219 {
220 dpsnprintf (name, sizeof(name), "%s_%i", loadmodel->name, i);
221 dpsnprintf (fogname, sizeof(fogname), "%s_%ifog", loadmodel->name, i);
222 }
223 if (!(skinframe = R_SkinFrame_LoadExternal(name, texflags | TEXF_COMPRESS, false, false)))
224 {
225 unsigned char *pixels = (unsigned char *) Mem_Alloc(loadmodel->mempool, width*height*4);
226 if (version == SPRITE32_VERSION)
227 {
228 for (x = 0;x < width*height;x++)
229 {
230 pixels[x*4+2] = datapointer[x*4+0];
231 pixels[x*4+1] = datapointer[x*4+1];
232 pixels[x*4+0] = datapointer[x*4+2];
233 pixels[x*4+3] = datapointer[x*4+3];
234 }
235 }
236 else //if (version == SPRITEHL_VERSION || version == SPRITE_VERSION)
237 Image_Copy8bitBGRA(datapointer, pixels, width*height, palette ? palette : palette_bgra_transparent);
238 skinframe = R_SkinFrame_LoadInternalBGRA(name, texflags, pixels, width, height, 0, 0, 0, false);
239 // texflags |= TEXF_COMPRESS;
240 Mem_Free(pixels);
241 }
242 }
243 if (skinframe == NULL)
244 skinframe = R_SkinFrame_LoadMissing();
245 Mod_SpriteSetupTexture(&loadmodel->data_textures[realframes], skinframe, fullbright, additive);
246 }
247
248 if (version == SPRITE32_VERSION)
249 datapointer += width * height * 4;
250 else //if (version == SPRITE_VERSION || version == SPRITEHL_VERSION)
251 datapointer += width * height;
252 realframes++;
253 }
254 }
255
256 modelradius = sqrt(modelradius);
257 for (i = 0;i < 3;i++)
258 {
259 loadmodel->normalmins[i] = loadmodel->yawmins[i] = loadmodel->rotatedmins[i] = -modelradius;
260 loadmodel->normalmaxs[i] = loadmodel->yawmaxs[i] = loadmodel->rotatedmaxs[i] = modelradius;
261 }
262 loadmodel->radius = modelradius;
263 loadmodel->radius2 = modelradius * modelradius;
264}

266void Mod_IDSP_Load(model_t *mod, void *buffer, void *bufferend)
267{
268 int version;
269 const unsigned char *datapointer;
270
271 datapointer = (unsigned char *)buffer;
272
273 loadmodel->modeldatatypestring = "SPR1";
274
275 loadmodel->type = mod_sprite;
276
277 loadmodel->Draw = R_Model_Sprite_Draw;
278 loadmodel->DrawDepth = NULL;
279 loadmodel->DrawLight = NULL;
280
281 version = LittleLong(((dsprite_t *)buffer)->version);
282 if (version == SPRITE_VERSION || version == SPRITE32_VERSION)
283 {
284 dsprite_t *pinqsprite;
285
286 pinqsprite = (dsprite_t *)datapointer;
287 datapointer += sizeof(dsprite_t);
288
289 loadmodel->numframes = LittleLong (pinqsprite->numframes);
290 loadmodel->sprite.sprnum_type = LittleLong (pinqsprite->type);
291 loadmodel->synctype = (synctype_t)LittleLong (pinqsprite->synctype);
292
293 Mod_Sprite_SharedSetup(datapointer, LittleLong (pinqsprite->version), NULL, false);
294 }
295 else if (version == SPRITEHL_VERSION)
296 {
297 int i, rendermode;
298 unsigned char palette[256][4];
299 const unsigned char *in;
300 dspritehl_t *pinhlsprite;
301 unsigned char *aligneddata;
302
303 pinhlsprite = (dspritehl_t *)datapointer;
304 datapointer += sizeof(dspritehl_t);
305
306 loadmodel->numframes = LittleLong (pinhlsprite->numframes);
307 loadmodel->sprite.sprnum_type = LittleLong (pinhlsprite->type);
308 loadmodel->synctype = (synctype_t)LittleLong (pinhlsprite->synctype);
309 rendermode = pinhlsprite->rendermode;
310
311 in = datapointer;
312 datapointer += 2;
313 i = in[0] + in[1] * 256;
314 if (i != 256)
315 Host_Error ("Mod_IDSP_Load: unexpected number of palette colors %i (should be 256)", i);
316 in = datapointer;
317 datapointer += 768;
318 switch(rendermode)
319 {
320 case SPRHL_OPAQUE:
321 for (i = 0;i < 256;i++)
322 {
323 palette[i][2] = in[i*3+0];
324 palette[i][1] = in[i*3+1];
325 palette[i][0] = in[i*3+2];
326 palette[i][3] = 255;
327 }
328 break;
329 case SPRHL_ADDITIVE:
330 for (i = 0;i < 256;i++)
331 {
332 palette[i][2] = in[i*3+0];
333 palette[i][1] = in[i*3+1];
334 palette[i][0] = in[i*3+2];
335 palette[i][3] = 255;
336 }
337 // also passes additive == true to Mod_Sprite_SharedSetup
338 break;
339 case SPRHL_INDEXALPHA:
340 for (i = 0;i < 256;i++)
341 {
342 palette[i][2] = in[765];
343 palette[i][1] = in[766];
344 palette[i][0] = in[767];
345 palette[i][3] = i;
346 in += 3;
347 }
348 break;
349 case SPRHL_ALPHATEST:
350 for (i = 0;i < 256;i++)
351 {
352 palette[i][2] = in[i*3+0];
353 palette[i][1] = in[i*3+1];
354 palette[i][0] = in[i*3+2];
355 palette[i][3] = 255;
356 }
357 palette[255][0] = palette[255][1] = palette[255][2] = palette[255][3] = 0;
358 // should this use alpha test or alpha blend? (currently blend)
359 break;
360 default:
361 Host_Error("Mod_IDSP_Load: unknown texFormat (%i, should be 0, 1, 2, or 3)", i);
362 return;
363 }
364
365 // the above datapointer arithmetic causes misaligned access
366 aligneddata = (unsigned char *)Mem_Alloc(tempmempool, (unsigned char *)bufferend - datapointer);
367 memcpy(aligneddata, datapointer, (unsigned char *)bufferend - datapointer);
368 Mod_Sprite_SharedSetup(aligneddata, LittleLong (pinhlsprite->version), (unsigned int *)(&palette[0][0]), rendermode == SPRHL_ADDITIVE);
369 Mem_Free(aligneddata);
370 }
371 else
372 Host_Error("Mod_IDSP_Load: %s has wrong version number (%i). Only %i (quake), %i (HalfLife), and %i (sprite32) supported",
373 loadmodel->name, version, SPRITE_VERSION, SPRITEHL_VERSION, SPRITE32_VERSION);
374
375 // TODO: Note that isanimated only means whether vertices change due to
376 // the animation. This may happen due to sprframe parameters changing.
377 // Mere texture chanegs OTOH shouldn't require isanimated to be 1.
378 loadmodel->surfmesh.isanimated = loadmodel->numframes > 1 || (loadmodel->animscenes && loadmodel->animscenes[0].framecount > 1);
379}

383{
384 int i, version;
385 qbool fullbright;
386 const dsprite2_t *pinqsprite;
387 skinframe_t *skinframe;
388 float modelradius;

398

400
401 version = LittleLong(pinqsprite->version);
402 if (version != SPRITE2_VERSION)
403 Host_Error("Mod_IDS2_Load: %s has wrong version number (%i should be 2 (quake 2)", loadmodel->name, version);
404
405 loadmodel->numframes = LittleLong (pinqsprite->numframes);
406 if (loadmodel->numframes < 1)
407 Host_Error ("Mod_IDS2_Load: Invalid # of frames: %d", loadmodel->numframes);
408 loadmodel->sprite.sprnum_type = SPR_VP_PARALLEL;

411 // LadyHavoc: hack to allow sprites to be non-fullbright

422
423 modelradius = 0;
424 for (i = 0;i < loadmodel->numframes;i++)
425 {

431 loadmodel->animscenes[i].firstframe = i;
432 loadmodel->animscenes[i].framecount = 1;
433 loadmodel->animscenes[i].framerate = 10;
434 loadmodel->animscenes[i].loop = true;
435

437

444

448 sprframe->right = -origin[0] + width;
449 sprframe->up = origin[1];
450 sprframe->down = origin[1] - height;
451
452 x = (int)max(sprframe->left * sprframe->left, sprframe->right * sprframe->right);
453 y = (int)max(sprframe->up * sprframe->up, sprframe->down * sprframe->down);
454 if (modelradius < x + y)
455 modelradius = x + y;
456 }
457
458 if (cls.state != ca_dedicated)
459 {
460 for (i = 0;i < loadmodel->numframes;i++)
461 {
462 const dsprite2frame_t *pinframe;
463 pinframe = &pinqsprite->frames[i];
464 if (!(skinframe = R_SkinFrame_LoadExternal(pinframe->name, texflags, false, false)))
465 {
466 Con_Printf(CON_ERROR "Mod_IDS2_Load: failed to load %s", pinframe->name);
467 skinframe = R_SkinFrame_LoadMissing();
468 }
469 Mod_SpriteSetupTexture(&loadmodel->data_textures[i], skinframe, fullbright, false);
470 }
471 }
472
473 modelradius = sqrt(modelradius);
474 for (i = 0;i < 3;i++)
475 {
476 loadmodel->normalmins[i] = loadmodel->yawmins[i] = loadmodel->rotatedmins[i] = -modelradius;
477 loadmodel->normalmaxs[i] = loadmodel->yawmaxs[i] = loadmodel->rotatedmaxs[i] = modelradius;
478 }
479 loadmodel->radius = modelradius;
480 loadmodel->radius2 = modelradius * modelradius;
481
482 // TODO: Note that isanimated only means whether vertices change due to
483 // the animation. This may happen due to sprframe parameters changing.
484 // Mere texture chanegs OTOH shouldn't require isanimated to be 1.
485 loadmodel->surfmesh.isanimated = loadmodel->numframes > 1 || (loadmodel->animscenes && loadmodel->animscenes[0].framecount > 1);
486}
