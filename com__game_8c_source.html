92void COM_InitGameType (void)
93{
94 char name [MAX_OSPATH];
95 int i;
96 int index = 0;
97
98#ifdef FORCEGAME
99 COM_ToLowerString(FORCEGAME, name, sizeof (name));
100#else
101 // check executable filename for keywords, but do it SMARTLY - only check the last path element
102 FS_StripExtension(FS_FileWithoutPath(sys.argv[0]), name, sizeof (name));
103 COM_ToLowerString(name, name, sizeof (name));
104#endif
105 for (i = 1;i < (int)(sizeof (gamemode_info) / sizeof (gamemode_info[0]));i++)
106 if (gamemode_info[i].prog_name && gamemode_info[i].prog_name[0] && strstr (name, gamemode_info[i].prog_name))
107 index = i;
108
109 // check commandline options for keywords
110 for (i = 0;i < (int)(sizeof (gamemode_info) / sizeof (gamemode_info[0]));i++)
111 if (Sys_CheckParm (gamemode_info[i].cmdline))
112 index = i;
113
114 com_startupgamemode = gamemode_info[index].mode;
115 com_startupgamegroup = gamemode_info[index].group;
116 COM_SetGameType(index);
117}

119void COM_ChangeGameTypeForGameDirs(void)
120{
121 unsigned i, gamemode_count = sizeof(gamemode_info) / sizeof(gamemode_info[0]);
122 int index = -1;
123 // this will not not change the gamegroup
124
125 // first check if a base game (single gamedir) matches
126 for (i = 0; i < gamemode_count; i++)
127 {
128 if (gamemode_info[i].group == com_startupgamegroup && !(gamemode_info[i].gamedirname2 && gamemode_info[i].gamedirname2[0]))
129 {
130 index = i;
131 break;
132 }
133 }
134 // now that we have a base game, see if there is a derivative game matching the startup one (two gamedirs)
135 // bones_was_here: this prevents a Quake expansion (eg Rogue) getting switched to Quake,
136 // and its gamedirname2 (eg "rogue") being lost from the search path, when adding a miscellaneous gamedir.
137 for (i = 0; i < gamemode_count; i++)
138 {
139 if (gamemode_info[i].group == com_startupgamegroup && (gamemode_info[i].gamedirname2 && gamemode_info[i].gamedirname2[0]) && gamemode_info[i].mode == com_startupgamemode)
140 {
141 index = i;
142 break;
143 }
144 }
145 // also see if the first gamedir (from -game parm or gamedir command) matches a derivative game (two/three gamedirs)
146 if (fs_numgamedirs)
147 {
148 for (i = 0; i < gamemode_count; i++)
149 {
150 if (gamemode_info[i].group == com_startupgamegroup && (gamemode_info[i].gamedirname2 && gamemode_info[i].gamedirname2[0]) && !strcasecmp(fs_gamedirs[0], gamemode_info[i].gamedirname2))
151 {
152 index = i;
153 break;
154 }
155 }
156 }
157 // we now have a good guess at which game this is meant to be...
158 if (index >= 0 && gamemode != gamemode_info[index].mode)
159 COM_SetGameType(index);
160}

162static void COM_SetGameType(int index)
163{
164 static char gamenetworkfilternamebuffer[64];
165 int i, t;
166 if (index < 0 || index >= (int)(sizeof (gamemode_info) / sizeof (gamemode_info[0])))
167 index = 0;
168 gamemode = gamemode_info[index].mode;
169 gamename = gamemode_info[index].gamename;
170 gamenetworkfiltername = gamemode_info[index].gamenetworkfiltername;
171 gamedirname1 = gamemode_info[index].gamedirname1;
172 gamedirname2 = gamemode_info[index].gamedirname2;
173 gamescreenshotname = gamemode_info[index].gamescreenshotname;
174 gameuserdirname = gamemode_info[index].gameuserdirname;
175
176 if (gamemode == com_startupgamemode)
177 {
178 if((t = Sys_CheckParm("-customgamename")) && t + 1 < sys.argc)
179 gamename = gamenetworkfiltername = sys.argv[t+1];
180 if((t = Sys_CheckParm("-customgamenetworkfiltername")) && t + 1 < sys.argc)
181 gamenetworkfiltername = sys.argv[t+1];
182 if((t = Sys_CheckParm("-customgamedirname1")) && t + 1 < sys.argc)
183 gamedirname1 = sys.argv[t+1];
184 if((t = Sys_CheckParm("-customgamedirname2")) && t + 1 < sys.argc)
185 gamedirname2 = *sys.argv[t+1] ? sys.argv[t+1] : NULL;
186 if((t = Sys_CheckParm("-customgamescreenshotname")) && t + 1 < sys.argc)
187 gamescreenshotname = sys.argv[t+1];
188 if((t = Sys_CheckParm("-customgameuserdirname")) && t + 1 < sys.argc)
189 gameuserdirname = sys.argv[t+1];
190 }
191
192 if (gamedirname2 && gamedirname2[0])
193 Con_Printf("Game is %s using base gamedirs %s %s", gamename, gamedirname1, gamedirname2);
194 else
195 Con_Printf("Game is %s using base gamedir %s", gamename, gamedirname1);
196 for (i = 0;i < fs_numgamedirs;i++)
197 {
198 if (i == 0)
199 Con_Printf(", with mod gamedirs");
200 Con_Printf(" %s", fs_gamedirs[i]);
201 }
202 Con_Printf("\n");
203
204 if (strchr(gamenetworkfiltername, ' '))
205 {
206 char *s;
207 // if there are spaces in the game's network filter name it would
208 // cause parse errors in getservers in dpmaster, so we need to replace
209 // them with _ characters
210 dp_strlcpy(gamenetworkfilternamebuffer, gamenetworkfiltername, sizeof(gamenetworkfilternamebuffer));
211 while ((s = strchr(gamenetworkfilternamebuffer, ' ')) != NULL)
212 *s = '_';
213 gamenetworkfiltername = gamenetworkfilternamebuffer;
214 }
215
216 Con_Printf("gamename for server filtering: %s\n", gamenetworkfiltername);
217}
